chmod o+x /путь/к/папке/или/файлу 
	- открыть доступ
chmod o-x /путь/к/папке/или/файлу
	- закрыть доступ

sudo lsof -i :<номер_порта> 
	- инфо по процессу занимающего указанный порт

history
	- история введённых команд
!<номер_команды>
	- запустить команду
!!
	- запустить последнюю введённую команду
!<текст>
	- запустить последнюю совпадающую с текстом команду

dpkg --list
	- список установленных пакетов

sudo apt-get remove имя_пакета
	- удалить пакет
sudo apt-get purge имя_пакета
	- удалить пакет и его конфигурационные файлы
sudo apt-get autoremove
	- удаляет неиспользуемые зависимости
